Mapa básico

<html>
		    <head>
		        <link rel="stylesheet"  
			href="http://dev.openlayers.org/examples/style.css"  
			type="text/css">
		        <script   
			src="http://openlayers.org/api/2.13.1/OpenLayers.js"></script>
		        <style>.olMap {width:512px;height:256px;}</style>
		    </head>
		    <body>
		        <div id="map-id"></div>
		        <script>
		            var map=new OpenLayers.Map("map-id");
		            map.addLayer(new OpenLayers.Layer.OSM());
		            map.zoomToMaxExtent();
		        </script>
		    </body>
</html>

Mapa básico

<html>
		    <head>
		        <link rel="stylesheet"  
			href="http://dev.openlayers.org/examples/style.css"  
			type="text/css">
		        <script   
			src="http://openlayers.org/api/2.13.1/OpenLayers.js"></script>
		        <style>.olMap {width:512px;height:256px;}</style>
		    </head>
		    <body>
		        <div id="map2"></div>
		        <script>
		            var map=new OpenLayers.Map("map2");
		            map.addLayer(new OpenLayers.Layer.OSM());
		            var bounds = new OpenLayers.Bounds();
		            bounds.extend(new OpenLayers.LonLat(-5, 35));
		            bounds.extend(new OpenLayers.LonLat(-7, 40));
		            map.zoomToExtent(bounds, true);
		        </script>
		    </body>
</html>
